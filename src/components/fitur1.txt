import { useState } from 'react' // 1. Import useState

function App() {
  // --- BAGIAN 1: DATA (STATE) ---
  
  // State untuk menyimpan daftar tugas (awalnya array kosong)
  const [todos, setTodos] = useState([]); 
  
  // State untuk menyimpan teks yang sedang diketik di input
  const [input, setInput] = useState(""); 

  // --- BAGIAN 2: LOGIKA (INTERAKSI) ---

  // Fungsi yang jalan saat tombol "Tambah" diklik
  const tambahTodo = () => {
    if (input.trim() === "") return; // Jangan tambah kalau kosong

    // Buat object todo baru
    const todoBaru = {
      id: Date.now(), // ID unik menggunakan waktu sekarang
      text: input
    };

    // Update daftar todos: Ambil semua todo lama (...todos), tambah yang baru
    setTodos([...todos, todoBaru]);
    
    // Kosongkan kolom input setelah menambah
    setInput("");
  };

  // --- BAGIAN 3: TAMPILAN (UI) ---
  return (
    <div className="min-h-screen bg-gray-100 flex items-center justify-center p-5">
      <div className="bg-white p-6 rounded-lg shadow-lg w-full max-w-md">
        <h1 className="text-2xl font-bold mb-4 text-center text-gray-800">Todo List Saya</h1>
        
        {/* Area Input */}
        <div className="flex gap-2 mb-4">
          <input 
            type="text" 
            className="border p-2 rounded w-full focus:outline-blue-500"
            placeholder="Mau ngapain hari ini?"
            value={input} // Kunci input dengan state 'input'
            onChange={(e) => setInput(e.target.value)} // Saat ngetik, update state 'input'
          />
          <button 
            onClick={tambahTodo} // Saat klik, jalankan fungsi tambahTodo
            className="bg-blue-500 text-white px-4 py-2 rounded hover:bg-blue-600 transition"
          >
            Tambah
          </button>
        </div>

        {/* Daftar Tugas */}
        <ul className="space-y-2">
          {todos.map((todo) => (
            <li key={todo.id} className="bg-gray-50 p-3 rounded border flex justify-between">
              {todo.text}
            </li>
          ))}
        </ul>
        
        {todos.length === 0 && <p className="text-gray-400 text-center">Belum ada tugas.</p>}
      </div>
    </div>
  )
}

export default App
